## 🛠️ Jouet — 45 min

### Objectif
Implémenter un mini-LSM-tree :
1. Memtable (dictionnaire Python trié)
2. Flush vers SSTable (fichier texte trié par clé)
3. Recherche dans N SSTables
4. Compaction par fusion (merge sort de 2 SSTables)

### Crée `code/01-storage-retrieval/lsm.py`

```python
"""
Mini-LSM-tree : memtable + SSTables + compaction.
Simplifié (pas de WAL, pas de filtre de Bloom, pas de niveaux).
"""
import os
import json
from pathlib import Path
from itertools import islice


TOMBSTONE = "__DELETED__"


class SSTable:
    """Fichier immuable avec clés triées, format : ligne par entrée 'clé\tvaleur'."""

    def __init__(self, path: Path):
        self.path = path

    def get(self, target_key: str) -> str | None:
        """Recherche linéaire (simplifiée — en production : index creux + saut)."""
        with open(self.path) as f:
            for line in f:
                key, value = line.rstrip("\n").split("\t", 1)
                if key == target_key:
                    return None if value == TOMBSTONE else value
                if key > target_key:
                    break  # clés triées → on peut arrêter
        return None

    @classmethod
    def flush(cls, path: Path, entries: dict[str, str]) -> "SSTable":
        """Écrit un dictionnaire trié sur disque."""
        path.parent.mkdir(parents=True, exist_ok=True)
        with open(path, "w") as f:
            for key in sorted(entries.keys()):
                f.write(f"{key}\t{entries[key]}\n")
        return cls(path)

    @classmethod
    def merge(cls, out_path: Path, tables: list["SSTable"]) -> "SSTable":
        """Fusionne N SSTables triées → 1 SSTable (merge sort).
        La table la plus récente (dernier index) a priorité."""
        iterators = [open(t.path) for t in tables]
        # Lire une ligne de chaque itérateur
        lines = []
        for i, it in enumerate(iterators):
            line = next(iter(it), None)
            if line:
                key, val = line.rstrip("\n").split("\t", 1)
                lines.append((key, val, i, it))

        out_path.parent.mkdir(parents=True, exist_ok=True)
        seen: dict[str, tuple[str, int]] = {}  # key → (value, table_index)

        # Collecte toutes les entrées
        for t_idx, table in enumerate(tables):
            with open(table.path) as f:
                for line in f:
                    key, val = line.rstrip("\n").split("\t", 1)
                    # Table plus récente = index plus élevé → priorité
                    if key not in seen or t_idx > seen[key][1]:
                        seen[key] = (val, t_idx)

        with open(out_path, "w") as f:
            for key in sorted(seen.keys()):
                val, _ = seen[key]
                if val != TOMBSTONE:  # supprimer les entrées marquées
                    f.write(f"{key}\t{val}\n")

        for it in iterators:
            it.close()
        return cls(out_path)

    def size(self) -> int:
        return self.path.stat().st_size


class MiniLSM:
    def __init__(self, data_dir: str = "/tmp/minilsm"):
        self.data_dir = Path(data_dir)
        self.data_dir.mkdir(parents=True, exist_ok=True)
        self.memtable: dict[str, str] = {}
        self.memtable_limit = 5  # petit pour la démo (en prod : ~64MB)
        self.sstables: list[SSTable] = []
        self._load_existing_sstables()

    def _load_existing_sstables(self) -> None:
        paths = sorted(self.data_dir.glob("sst_*.txt"))
        self.sstables = [SSTable(p) for p in paths]

    def _flush_memtable(self) -> None:
        if not self.memtable:
            return
        idx = len(self.sstables)
        path = self.data_dir / f"sst_{idx:04d}.txt"
        sst = SSTable.flush(path, self.memtable)
        self.sstables.append(sst)
        self.memtable.clear()
        print(f"  [flush] → {path.name} ({sst.size()} octets)")

    def put(self, key: str, value: str) -> None:
        self.memtable[key] = value
        if len(self.memtable) >= self.memtable_limit:
            self._flush_memtable()

    def delete(self, key: str) -> None:
        self.memtable[key] = TOMBSTONE
        if len(self.memtable) >= self.memtable_limit:
            self._flush_memtable()

    def get(self, key: str) -> str | None:
        # 1. Cherche dans la memtable (la plus récente)
        if key in self.memtable:
            val = self.memtable[key]
            return None if val == TOMBSTONE else val
        # 2. Cherche dans les SSTables du plus récent au plus ancien
        for sst in reversed(self.sstables):
            val = sst.get(key)
            if val is not None:
                return val
        return None

    def compact(self) -> None:
        if len(self.sstables) < 2:
            return
        # Flush la memtable d'abord
        self._flush_memtable()
        out_path = self.data_dir / "sst_compacted.txt"
        print(f"  [compact] fusion de {len(self.sstables)} SSTables…")
        merged = SSTable.merge(out_path, self.sstables)
        # Supprimer les anciennes SSTables
        for sst in self.sstables:
            sst.path.unlink(missing_ok=True)
        final_path = self.data_dir / f"sst_{0:04d}.txt"
        out_path.rename(final_path)
        self.sstables = [SSTable(final_path)]
        print(f"  [compact] → {final_path.name} ({merged.size()} octets)")

    def stats(self) -> dict:
        return {
            "memtable_entries": len(self.memtable),
            "sstables_count": len(self.sstables),
            "total_sst_bytes": sum(s.size() for s in self.sstables),
        }


if __name__ == "__main__":
    import shutil
    shutil.rmtree("/tmp/minilsm", ignore_errors=True)

    lsm = MiniLSM()

    # Écriture : le flush se déclenche toutes les 5 entrées
    print("── Écriture (flush auto toutes les 5 entrées) ──")
    for i in range(12):
        lsm.put(f"key:{i:03d}", f"value_{i}")

    print(f"Stats : {lsm.stats()}")

    # Lecture depuis la memtable
    print(f"\nget(key:011) → {lsm.get('key:011')}")  # memtable
    # Lecture depuis SSTable
    print(f"get(key:003) → {lsm.get('key:003')}")  # SSTable L0
    print(f"get(key:999) → {lsm.get('key:999')}")  # None

    # Mise à jour (crée une nouvelle version dans memtable)
    lsm.put("key:003", "updated_3")
    print(f"get(key:003) après update → {lsm.get('key:003')}")

    # Suppression
    lsm.delete("key:005")
    print(f"get(key:005) après delete → {lsm.get('key:005')}")  # None

    # Compaction
    print(f"\n── Compaction ──")
    before = lsm.stats()["total_sst_bytes"]
    lsm.compact()
    after = lsm.stats()["total_sst_bytes"]
    print(f"Avant : {before}B · Après : {after}B")
    print(f"get(key:003) après compaction → {lsm.get('key:003')}")  # updated_3
    print(f"get(key:005) après compaction → {lsm.get('key:005')}")  # None
```
