```python
"""
Magasin clé-valeur en mémoire avec MVCC (Multi-Version Concurrency Control).
Implémente snapshot isolation :
  - Chaque écriture crée une nouvelle version (tx_id, value)
  - Chaque transaction lit uniquement les versions committées avant son début
  - Readers don't block writers, writers don't block readers
  - Détecte les conflits write-write (mais PAS write skew — voir anomalies.py)
"""
import threading
from dataclasses import dataclass, field


# ═══════════════════════════════════════════════════════════════════════════════
# Moteur MVCC
# ═══════════════════════════════════════════════════════════════════════════════

@dataclass
class Version:
    """Une version d'une valeur (créée par une transaction)."""
    tx_id: int
    value: str | None    # None = tombstone (DELETE)
    committed: bool = False
    aborted: bool = False


class MVCCStore:
    """
    Magasin clé-valeur avec Multi-Version Concurrency Control.
    Chaque clé a une liste de versions ordonnées par tx_id croissant.
    """

    def __init__(self):
        self._versions: dict[str, list[Version]] = {}  # clé → liste de versions
        self._tx_counter = 0
        self._committed_txs: set[int] = set()          # tx_ids committés
        self._lock = threading.Lock()

    def _next_tx_id(self) -> int:
        with self._lock:
            self._tx_counter += 1
            return self._tx_counter

    def begin(self) -> "Transaction":
        """Démarrer une nouvelle transaction avec un snapshot de l'état courant."""
        tx_id = self._next_tx_id()
        # Le snapshot = ensemble des tx committées AU MOMENT où cette txn commence
        with self._lock:
            snapshot = frozenset(self._committed_txs)
        return Transaction(tx_id=tx_id, store=self, snapshot=snapshot)

    def _write(self, key: str, value: str | None, tx_id: int) -> None:
        """Écriture interne (appelée par Transaction)."""
        with self._lock:
            if key not in self._versions:
                self._versions[key] = []
            self._versions[key].append(Version(tx_id=tx_id, value=value))

    def _read(self, key: str, snapshot: frozenset[int]) -> str | None:
        """
        Lire la version la plus récente visible pour ce snapshot.
        Visible = committée ET tx_id dans le snapshot (committée avant le début de cette txn).
        """
        with self._lock:
            versions = self._versions.get(key, [])
        # Parcourir du plus récent au plus ancien
        for v in reversed(versions):
            if v.tx_id in snapshot and v.committed and not v.aborted:
                return v.value  # None = DELETE
        return None  # clé inexistante

    def _commit(self, tx_id: int) -> None:
        with self._lock:
            # Marquer toutes les versions de cette transaction comme committées
            for versions in self._versions.values():
                for v in versions:
                    if v.tx_id == tx_id:
                        v.committed = True
            self._committed_txs.add(tx_id)

    def _abort(self, tx_id: int) -> None:
        with self._lock:
            for versions in self._versions.values():
                for v in versions:
                    if v.tx_id == tx_id:
                        v.aborted = True

    def debug_versions(self, key: str) -> list[tuple]:
        """Afficher toutes les versions d'une clé (pour débogage)."""
        with self._lock:
            versions = self._versions.get(key, [])
            return [(v.tx_id, v.value, "committed" if v.committed else "aborted" if v.aborted else "pending")
                    for v in versions]


class Transaction:
    """Transaction MVCC avec snapshot isolation."""

    def __init__(self, tx_id: int, store: MVCCStore, snapshot: frozenset[int]):
        self.tx_id = tx_id
        self._store = store
        self._snapshot = frozenset(list(snapshot) + [tx_id])  # voit ses propres écritures
        self._written_keys: set[str] = set()
        self._active = True

    def read(self, key: str) -> str | None:
        """Lire la valeur visible dans ce snapshot."""
        assert self._active, "Transaction terminée"
        return self._store._read(key, self._snapshot)

    def write(self, key: str, value: str) -> None:
        """Écrire une nouvelle version (invisible jusqu'au commit)."""
        assert self._active, "Transaction terminée"
        self._store._write(key, value, self.tx_id)
        self._written_keys.add(key)

    def delete(self, key: str) -> None:
        """Supprimer une clé (tombstone)."""
        assert self._active, "Transaction terminée"
        self._store._write(key, None, self.tx_id)
        self._written_keys.add(key)

    def commit(self) -> None:
        assert self._active, "Transaction terminée"
        self._store._commit(self.tx_id)
        self._active = False
        print(f"  [T{self.tx_id}] COMMIT ✓ (écrit sur : {self._written_keys or '∅'})")

    def abort(self) -> None:
        assert self._active, "Transaction terminée"
        self._store._abort(self.tx_id)
        self._active = False
        print(f"  [T{self.tx_id}] ROLLBACK")

    def __repr__(self) -> str:
        return f"Transaction(tx_id={self.tx_id}, snapshot={sorted(self._snapshot)})"


# ═══════════════════════════════════════════════════════════════════════════════
# Démonstrations
# ═══════════════════════════════════════════════════════════════════════════════

def demo_snapshot_isolation() -> None:
    """Une transaction voit un snapshot cohérent même si une autre commite entre-temps."""
    print("\n═══ Démonstration 1 : Snapshot Isolation ═══")
    store = MVCCStore()

    # Transaction initiale : écrire une valeur
    t0 = store.begin()
    t0.write("balance:alice", "100")
    t0.commit()

    # T1 commence → snapshot = {0} (seulement t0)
    t1 = store.begin()
    print(f"  T1 commence avec snapshot={sorted(t1._snapshot)}")

    # T2 modifie alice et commite
    t2 = store.begin()
    t2.write("balance:alice", "50")
    t2.commit()
    print(f"  T2 commite (balance:alice → 50)")

    # T1 lit → doit voir 100 (son snapshot ne contient pas T2)
    v1 = t1.read("balance:alice")
    print(f"  T1 lit balance:alice → {v1!r} (attendu: '100')")
    assert v1 == "100", f"✗ Snapshot isolation violé ! Vu: {v1}"
    print("  ✓ T1 voit le snapshot cohérent (ne voit pas la modification de T2)")
    t1.commit()

    print(f"\n  Versions de 'balance:alice' :")
    for v in store.debug_versions("balance:alice"):
        print(f"    tx_id={v[0]} value={v[1]!r} {v[2]}")


def demo_no_blocking() -> None:
    """Readers ne bloquent pas writers et vice-versa."""
    print("\n═══ Démonstration 2 : Readers don't block writers ═══")
    store = MVCCStore()
    t0 = store.begin()
    t0.write("data", "v1")
    t0.commit()

    results = []

    def long_read():
        t = store.begin()
        import time
        v = t.read("data")
        time.sleep(0.1)          # lecture longue (simule analytics)
        v_after = t.read("data") # relit dans le même snapshot
        t.commit()
        results.append((v, v_after))

    def concurrent_write():
        import time
        time.sleep(0.02)  # démarre pendant que long_read dort
        t = store.begin()
        t.write("data", "v2")
        t.commit()
        print("  Écriture v2 committée pendant la lecture longue")

    import threading
    r = threading.Thread(target=long_read)
    w = threading.Thread(target=concurrent_write)
    r.start(); w.start()
    r.join(); w.join()

    v, v_after = results[0]
    print(f"  Lecture longue : début={v!r} fin={v_after!r}")
    assert v == v_after == "v1", "✗ Snapshot inconsistant !"
    print("  ✓ Snapshot stable : v1 visible du début à la fin, même si v2 a été écrite entre-temps")


def demo_write_conflict_detection() -> None:
    """Détecter un conflit write-write simple (même clé, même tx)."""
    print("\n═══ Démonstration 3 : Conflit write-write (même clé) ═══")
    print("  Note : MVCC simple ne détecte PAS le write skew (voir anomalies.py)")
    store = MVCCStore()
    t0 = store.begin()
    t0.write("counter", "10")
    t0.commit()

    # T1 et T2 lisent, puis écrivent
    t1 = store.begin()
    t2 = store.begin()
    v1 = t1.read("counter")
    v2 = t2.read("counter")
    print(f"  T1 lit counter → {v1}, T2 lit counter → {v2}")

    t1.write("counter", str(int(v1) + 1))  # 11
    t1.commit()

    t2.write("counter", str(int(v2) + 1))  # aussi 11 → lost update !
    t2.commit()

    t3 = store.begin()
    final = t3.read("counter")
    t3.commit()
    print(f"  Valeur finale : {final} (attendu : 12 si sans conflit)")
    print(f"  → Lost update ! T1 et T2 ont tous les 2 écrit 11 (attendu 12)")
    print(f"  → Solution : SELECT FOR UPDATE ou atomic write (UPDATE counter = counter + 1)")


def main() -> None:
    demo_snapshot_isolation()
    demo_no_blocking()
    demo_write_conflict_detection()


if __name__ == "__main__":
    main()
```
