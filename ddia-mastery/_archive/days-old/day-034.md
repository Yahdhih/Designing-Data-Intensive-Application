```python
"""
Simulateur de quorum Dynamo-style (N/W/R configurable).
Démontre :
  - Pourquoi W+R>N garantit des lectures non-périmées
  - Pourquoi W+R<=N peut retourner une valeur périmée
  - L'effet des pannes sur la disponibilité (vs cohérence)
"""
import random
from dataclasses import dataclass, field


# ═══════════════════════════════════════════════════════════════════════════════
# Modèle de données
# ═══════════════════════════════════════════════════════════════════════════════

@dataclass
class VersionedValue:
    """Valeur avec numéro de version (remplace vector clocks pour la démo)."""
    value: str
    version: int


class QuorumNode:
    """Un nœud de stockage dans le cluster leaderless."""

    def __init__(self, node_id: int):
        self.node_id = node_id
        self._store: dict[str, VersionedValue] = {}
        self.available = True  # peut être mis à False pour simuler une panne

    def write(self, key: str, value: str, version: int) -> bool:
        """Retourne True si le nœud est disponible (écriture réussie)."""
        if not self.available:
            return False
        existing = self._store.get(key)
        # Règle LWW sur version : on prend la version la plus haute
        if existing is None or version > existing.version:
            self._store[key] = VersionedValue(value=value, version=version)
        return True

    def read(self, key: str) -> VersionedValue | None:
        """Retourne None si le nœud est indisponible."""
        if not self.available:
            return None
        return self._store.get(key)

    def __repr__(self) -> str:
        return f"Node({self.node_id}, {'UP' if self.available else 'DOWN'})"


# ═══════════════════════════════════════════════════════════════════════════════
# Cluster leaderless
# ═══════════════════════════════════════════════════════════════════════════════

class LeaderlessCluster:
    """
    Cluster sans leader avec quorum W/R/N configurable.
    Le coordinateur (client) envoie les requêtes à tous les nœuds
    et attend W (write) ou R (read) confirmations.
    """

    def __init__(self, n: int, w: int, r: int):
        assert w <= n and r <= n, "W et R doivent être <= N"
        self.N = n
        self.W = w
        self.R = r
        self.nodes = [QuorumNode(i) for i in range(n)]
        self._version_counter = 0

    def _available_nodes(self) -> list[QuorumNode]:
        return [nd for nd in self.nodes if nd.available]

    def write(self, key: str, value: str) -> tuple[bool, list[int]]:
        """
        Écriture sur W nœuds (choisis aléatoirement parmi les disponibles).
        Retourne (succès, indices des nœuds écrits).
        """
        self._version_counter += 1
        version = self._version_counter
        available = self._available_nodes()

        if len(available) < self.W:
            return False, []  # Quorum d'écriture non atteignable

        # Choisir W nœuds aléatoirement (comme le ferait un vrai coordinateur)
        chosen = random.sample(available, self.W)
        written_ids = []
        for nd in chosen:
            if nd.write(key, value, version):
                written_ids.append(nd.node_id)

        success = len(written_ids) >= self.W
        return success, written_ids

    def read(self, key: str) -> tuple[str | None, list[int], bool]:
        """
        Lecture depuis R nœuds, retourne la valeur avec la version la plus haute.
        Retourne (valeur, indices des nœuds lus, quorum_atteint).
        """
        available = self._available_nodes()

        if len(available) < self.R:
            return None, [], False  # Quorum de lecture non atteignable

        chosen = random.sample(available, self.R)
        responses: list[tuple[VersionedValue, int]] = []
        read_ids = []
        for nd in chosen:
            val = nd.read(key)
            if val is not None:
                responses.append((val, nd.node_id))
                read_ids.append(nd.node_id)

        if not responses:
            return None, read_ids, True

        # Sélectionner la version la plus haute (Dynamo: "last-write-wins" sur version)
        best_val, best_node = max(responses, key=lambda x: x[0].version)
        return best_val.value, read_ids, True

    def set_node_available(self, node_id: int, available: bool) -> None:
        self.nodes[node_id].available = available
        status = "UP" if available else "DOWN"
        print(f"  [Cluster] Node-{node_id} → {status}")


# ═══════════════════════════════════════════════════════════════════════════════
# Démonstrations
# ═══════════════════════════════════════════════════════════════════════════════

def demo_quorum_guarantees() -> None:
    """Démontrer que W+R>N garantit les lectures non-périmées."""
    print("\n═══ Démonstration 1 : W+R>N garantit la cohérence ═══")

    # N=3, W=2, R=2 → W+R=4>3 ✓
    cluster = LeaderlessCluster(n=3, w=2, r=2)
    print(f"  Config : N={cluster.N}, W={cluster.W}, R={cluster.R} → W+R={cluster.W+cluster.R}")
    print(f"  Chevauchement garanti : W+R-N = {cluster.W+cluster.R-cluster.N} nœuds communs")

    # 10 tests : write puis read → doit toujours voir la valeur récente
    stale_reads = 0
    for i in range(10):
        ok, written = cluster.write("key", f"v{i+1}")
        if not ok:
            continue
        value, read_from, _ = cluster.read("key")
        if value != f"v{i+1}":
            stale_reads += 1
            print(f"  ✗ Test {i+1}: écrit sur {written}, lu {value!r} depuis {read_from}")

    print(f"  Lectures périmées sur 10 tests : {stale_reads}/10")
    if stale_reads == 0:
        print("  ✓ W+R>N → aucune lecture périmée (comme attendu)")


def demo_stale_reads() -> None:
    """Démontrer que W+R<=N peut retourner des valeurs périmées."""
    print("\n═══ Démonstration 2 : W+R<=N → lectures périmées ═══")

    # N=3, W=1, R=1 → W+R=2 <= 3 ✗
    cluster = LeaderlessCluster(n=3, w=1, r=1)
    print(f"  Config : N={cluster.N}, W={cluster.W}, R={cluster.R} → W+R={cluster.W+cluster.R}")
    print(f"  Chevauchement garanti : W+R-N = {cluster.W+cluster.R-cluster.N} (négatif → aucune garantie !)")

    # 20 tests
    stale_reads = 0
    for i in range(20):
        ok, written = cluster.write("key", f"v{i+1}")
        if not ok:
            continue
        value, read_from, _ = cluster.read("key")
        if value != f"v{i+1}":
            stale_reads += 1

    print(f"  Lectures périmées sur 20 tests : {stale_reads}/20")
    if stale_reads > 0:
        print(f"  ✗ W+R≤N → {stale_reads} lectures périmées observées (comme prédit)")
    else:
        print("  (Par chance aucune périmée — rejoue plusieurs fois pour observer l'anomalie)")


def demo_availability_vs_consistency() -> None:
    """Montrer l'effet des pannes sur disponibilité vs cohérence."""
    print("\n═══ Démonstration 3 : Pannes → compromis disponibilité/cohérence ═══")

    # N=3, W=2, R=2 (QUORUM)
    cluster = LeaderlessCluster(n=3, w=2, r=2)

    # Écriture normale
    ok, _ = cluster.write("data", "valeur-initiale")
    print(f"  Écriture initiale : {'✓' if ok else '✗'}")

    # Simuler la panne de 1 nœud → QUORUM toujours atteignable
    print("\n  Panne d'1 nœud sur 3 :")
    cluster.set_node_available(0, False)
    ok_w, _ = cluster.write("data", "après-panne-1")
    val, _, ok_r = cluster.read("data")
    print(f"  Écriture : {'✓' if ok_w else '✗'}  |  Lecture : {val!r} {'✓' if ok_r else '✗'}")

    # Simuler la panne de 2 nœuds → QUORUM non atteignable (N=3, W=2 → besoin de 2 UP)
    print("\n  Panne de 2 nœuds sur 3 :")
    cluster.set_node_available(1, False)
    ok_w, _ = cluster.write("data", "après-panne-2")
    val, _, ok_r = cluster.read("data")
    print(f"  Écriture : {'✓' if ok_w else '✗ (quorum non atteint)'}  |  Lecture : {val!r} {'✓' if ok_r else '✗ (quorum non atteint)'}")

    print("\n  → Avec N=3, W=2, R=2 : tolère 1 panne, pas 2.")
    print("    Si on veut tolérer 2 pannes sur 3 → W=R=1 (mais perd la cohérence).")

    # Remettre les nœuds UP
    cluster.set_node_available(0, True)
    cluster.set_node_available(1, True)


def main() -> None:
    print("Simulateur de quorum Dynamo-style")
    demo_quorum_guarantees()
    demo_stale_reads()
    demo_availability_vs_consistency()

    print("\n═══ Résumé des configurations typiques Cassandra ═══")
    configs = [
        (3, 1, 1, "Cassandra ONE — haute disponibilité, cohérence éventuelle"),
        (3, 2, 2, "Cassandra QUORUM — compromis (tolère 1 panne)"),
        (3, 3, 1, "Écriture forte (ALL write) — tolère 0 panne en écriture"),
        (3, 1, 3, "Lecture forte (ALL read) — tolère 0 panne en lecture"),
    ]
    for n, w, r, desc in configs:
        guarantee = "✓" if w + r > n else "✗"
        print(f"  N={n}, W={w}, R={r} → W+R={'>' if w+r>n else '<='}{n} {guarantee} | {desc}")


if __name__ == "__main__":
    main()
```
