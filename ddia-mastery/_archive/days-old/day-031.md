```python
"""
Simulateur de réplication mono-leader.
Démontre 3 anomalies de lecture dues au lag de réplication asynchrone :
  1. Read-your-writes
  2. Monotonic reads (voyage dans le passé)
  3. Consistent prefix reads (ordre causal violé)
"""
import time
import threading
import random
from dataclasses import dataclass, field


# ═══════════════════════════════════════════════════════════════════════════════
# Modèle de données
# ═══════════════════════════════════════════════════════════════════════════════

@dataclass
class Entry:
    """Entrée dans le log de réplication."""
    key: str
    value: str
    seq: int   # numéro de séquence global (monotone, depuis le leader)
    ts: float = field(default_factory=time.time)


class Node:
    """Un nœud de stockage (leader ou follower)."""

    def __init__(self, name: str):
        self.name = name
        self._store: dict[str, Entry] = {}
        self._lock = threading.Lock()
        self._applied_seq: int = 0  # dernier seq appliqué

    def apply(self, entry: Entry) -> None:
        """Appliquer une entrée du log (écriture physique)."""
        with self._lock:
            self._store[entry.key] = entry
            self._applied_seq = max(self._applied_seq, entry.seq)

    def read(self, key: str) -> tuple[str | None, int]:
        """Lire la valeur + le seq qui l'a produite."""
        with self._lock:
            if key in self._store:
                e = self._store[key]
                return e.value, e.seq
            return None, 0

    def applied_seq(self) -> int:
        with self._lock:
            return self._applied_seq


# ═══════════════════════════════════════════════════════════════════════════════
# Cluster mono-leader
# ═══════════════════════════════════════════════════════════════════════════════

class SingleLeaderCluster:
    """
    1 leader + N followers.
    Réplication asynchrone : le leader confirme immédiatement,
    propage aux followers avec un délai artificiel.
    """

    def __init__(self, n_followers: int = 2, lag_range_ms: tuple[int, int] = (200, 800)):
        self.leader = Node("leader")
        self.followers = [
            Node(f"follower-{i + 1}") for i in range(n_followers)
        ]
        self._lag_range = lag_range_ms
        self._seq = 0
        self._lock = threading.Lock()

    def write(self, key: str, value: str) -> int:
        """
        Écriture sur le leader + réplication asynchrone.
        Retourne le numéro de séquence de l'écriture.
        """
        with self._lock:
            self._seq += 1
            seq = self._seq

        entry = Entry(key=key, value=value, seq=seq)
        self.leader.apply(entry)  # synchrone : leader confirme immédiatement

        # Réplication asynchrone vers les followers (avec lag)
        for follower in self.followers:
            lag = random.uniform(*self._lag_range) / 1000
            def _replicate(f=follower, e=entry, l=lag):
                time.sleep(l)
                f.apply(e)
            threading.Thread(target=_replicate, daemon=True).start()

        return seq

    def read_from_leader(self, key: str) -> str | None:
        value, _ = self.leader.read(key)
        return value

    def read_from_follower(self, follower_idx: int, key: str) -> str | None:
        """Lire depuis un follower spécifique (peut voir des données périmées)."""
        value, _ = self.followers[follower_idx].read(key)
        return value

    def read_monotonic(self, key: str, min_seq: int) -> tuple[str | None, str]:
        """
        Lecture monotone : sélectionner le follower qui a appliqué ≥ min_seq.
        Si aucun follower n'est à jour, lire depuis le leader.
        """
        for f in self.followers:
            if f.applied_seq() >= min_seq:
                value, _ = f.read(key)
                return value, f.name
        value, _ = self.leader.read(key)
        return value, "leader (fallback)"

    def read_your_writes(self, key: str, written_seq: int) -> tuple[str | None, str]:
        """
        Read-your-writes : garantir que la lecture retourne
        la valeur écrite au seq 'written_seq'.
        """
        return self.read_monotonic(key, written_seq)


# ═══════════════════════════════════════════════════════════════════════════════
# Démonstrations
# ═══════════════════════════════════════════════════════════════════════════════

def demo_read_your_writes(cluster: SingleLeaderCluster) -> None:
    print("\n═══ Anomalie 1 : Read-your-writes ═══")
    seq = cluster.write("profile:alice", "Alice Martin")
    print(f"  Alice écrit son profil (seq={seq}). Lecture 50ms après...")
    time.sleep(0.05)  # 50ms — moins que le lag minimum (200ms)

    # Sans garantie : lire depuis un follower peut donner None
    value = cluster.read_from_follower(0, "profile:alice")
    print(f"  Sans garantie (follower-1) : {value!r}")  # probablement None

    # Avec garantie read-your-writes : fallback vers le leader si nécessaire
    value, source = cluster.read_your_writes("profile:alice", seq)
    print(f"  Avec read-your-writes ({source}) : {value!r}")


def demo_monotonic_reads(cluster: SingleLeaderCluster) -> None:
    print("\n═══ Anomalie 2 : Monotonic reads ═══")
    seq = cluster.write("score:bob", "100")
    print(f"  Bob obtient le score 100 (seq={seq}). Lecture 50ms après...")
    time.sleep(0.05)

    # Deux lectures depuis 2 followers différents → résultats potentiellement différents
    v0 = cluster.read_from_follower(0, "score:bob")
    v1 = cluster.read_from_follower(1, "score:bob")
    print(f"  follower-1 : {v0!r}  |  follower-2 : {v1!r}")
    if v0 != v1:
        print("  ⚠ Monotonic reads violé : 2 lectures successives retournent des valeurs différentes !")
    else:
        print("  (Rejoue si les deux followers sont synchronisés — augmente le lag si nécessaire)")

    # Avec garantie : lire toujours depuis le même follower (hash par user_id)
    follower_idx = hash("bob") % len(cluster.followers)
    v_sticky = cluster.read_from_follower(follower_idx, "score:bob")
    print(f"  Avec stickiness (follower-{follower_idx + 1}) : {v_sticky!r}")


def demo_consistent_prefix_reads(cluster: SingleLeaderCluster) -> None:
    print("\n═══ Anomalie 3 : Consistent prefix reads ═══")
    # Simulation : 2 séquences causalement liées
    seq_q = cluster.write("qa:thread:1:question", "DDIA vaut-il la peine d'être lu ?")
    time.sleep(0.1)  # la réponse arrive après la question (causalité)
    seq_a = cluster.write("qa:thread:1:reponse", "Absolument !")
    print(f"  Question (seq={seq_q}) puis Réponse (seq={seq_a}) écrites sur le leader.")
    time.sleep(0.05)  # 50ms : les followers peuvent avoir un lag différent

    q = cluster.read_from_follower(0, "qa:thread:1:question")
    a = cluster.read_from_follower(0, "qa:thread:1:reponse")
    print(f"  follower-1 voit : Q={q!r}, R={a!r}")
    if q is None and a is not None:
        print("  ⚠ Consistent prefix reads violé : la réponse est visible avant la question !")
    else:
        print("  (Même follower → pas d'anomalie. L'anomalie apparaît si Q et R sont sur des partitions différentes.)")

    # Note : dans un vrai système partitionné, Q et R peuvent être sur
    # des partitions avec des followers ayant des lags différents.
    print("  → Solution : co-localiser Q et R dans la même partition (même clé de partition).")


def main() -> None:
    print("Initialisation du cluster : 1 leader + 2 followers, lag 200-800ms")
    cluster = SingleLeaderCluster(n_followers=2, lag_range_ms=(200, 800))

    demo_read_your_writes(cluster)
    demo_monotonic_reads(cluster)
    demo_consistent_prefix_reads(cluster)

    # Attendre que les followers se synchronisent
    print("\n⏳ Attente de la synchronisation des followers (1s)...")
    time.sleep(1.0)
    v, _ = cluster.leader.read("profile:alice")
    f0, _ = cluster.followers[0].read("profile:alice")
    f1, _ = cluster.followers[1].read("profile:alice")
    print(f"  leader: {v!r} | follower-1: {f0!r} | follower-2: {f1!r}")
    print("  → Cohérence éventuelle : tous les nœuds convergent vers la même valeur ✓")


if __name__ == "__main__":
    main()
```
