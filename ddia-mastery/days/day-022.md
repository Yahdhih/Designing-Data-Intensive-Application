### Étape 2 — Encoder / décoder avec le schéma v1 (10 min)

Crée `code/02-encoding/protobuf_demo.py` :

```python
"""
Démonstration Protobuf : encodage, taille, et évolution de schéma.
"""
import json, sys
sys.path.insert(0, "code/02-encoding")
from user_pb2 import User

# ── Encodage de base ──────────────────────────────────────────────────────────
alice_v1 = User(user_id=42, name="Alice")
encoded = alice_v1.SerializeToString()

print("═══ Encodage basique ═══")
print(f"  Proto bytes : {encoded.hex()}")
print(f"  Taille proto  : {len(encoded)} bytes")
json_eq = json.dumps({"user_id": 42, "name": "Alice"}).encode()
print(f"  Taille JSON   : {len(json_eq)} bytes")
print(f"  Ratio         : {len(json_eq) / len(encoded):.1f}× plus compact")

# ── Décodage ─────────────────────────────────────────────────────────────────
decoded = User()
decoded.ParseFromString(encoded)
print(f"\n  Décodé : user_id={decoded.user_id}, name='{decoded.name}'")
assert decoded.user_id == 42 and decoded.name == "Alice"
print("  ✓ Décodage correct")
```

### Étape 3 — Évolution du schéma (20 min)

**Ajoute un champ `email` (field tag 3) dans `user.proto` :**

```protobuf
syntax = "proto3";

package ddia;

message User {
  int32  user_id = 1;
  string name    = 2;
  string email   = 3;  // ← AJOUT : champ optionnel
}
```

Recompile :
```bash
python -m grpc_tools.protoc -I code/02-encoding \
    --python_out=code/02-encoding code/02-encoding/user.proto
```

**Ajoute ce bloc dans `protobuf_demo.py` :**

```python
from user_pb2 import User as UserV2

print("\n═══ Évolution de schéma ═══")

# Le message v2 inclut email
alice_v2 = UserV2(user_id=42, name="Alice", email="alice@example.com")
encoded_v2 = alice_v2.SerializeToString()
print(f"  Message v2 : {encoded_v2.hex()}")

# ── Forward compat : vieux schéma (simulé) lit message v2 ──
# Protobuf ignore les champs inconnus → field_id=3 ignoré par un décodeur v1
decoded_by_v1 = User()
try:
    # En proto3, les champs inconnus sont ignorés silencieusement
    decoded_by_v1.ParseFromString(encoded_v2)
    print(f"  Décodé par 'v1' : user_id={decoded_by_v1.user_id}, name='{decoded_by_v1.name}'")
    print(f"  → email ignoré silencieusement ✓ (forward compat)")
except Exception as e:
    print(f"  ✗ Erreur inattendue : {e}")

# ── Backward compat : nouveau schéma lit message v1 ──
msg_v1_bytes = User(user_id=42, name="Alice").SerializeToString()
decoded_v2_schema = UserV2()
decoded_v2_schema.ParseFromString(msg_v1_bytes)
print(f"\n  Message v1 décodé avec schéma v2 :")
print(f"  user_id={decoded_v2_schema.user_id}, name='{decoded_v2_schema.name}', email='{decoded_v2_schema.email}'")
print(f"  → email = '' (valeur par défaut proto3) ✓ (backward compat)")

# ── DANGER : que se passe-t-il si on change le type d'un champ existant ? ──
print("\n═══ DANGER : changer le TYPE d'un champ ═══")
# Si field_id=2 était int32 et devient string → corruption silencieuse
# → Toujours ajouter de NOUVEAUX champs avec de NOUVEAUX field tags
print("  Règle d'or : ne jamais réassigner ou changer le type d'un field tag existant.")
print("  Protobuf ne peut pas valider ça au runtime — corruption silencieuse garantie.")
```
