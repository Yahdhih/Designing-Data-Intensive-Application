### Le jouet

Crée `code/common/percentiles.py` :

```python
"""
Jouet : Calcul de percentiles sur des latences simulées.
Objectif : montrer pourquoi la moyenne ment.
"""
import random
import numpy as np

def generate_latencies(n_normal: int, n_slow: int,
                       normal_ms: float, slow_ms: float) -> list[float]:
    """Génère un mix de requêtes normales et lentes."""
    normal = [random.gauss(normal_ms, normal_ms * 0.1) for _ in range(n_normal)]
    slow   = [random.gauss(slow_ms,   slow_ms   * 0.2) for _ in range(n_slow)]
    return normal + slow

def compute_stats(latencies: list[float]) -> dict:
    """Calcule moyenne, médiane et percentiles."""
    arr = np.array(latencies)
    return {
        "n":       len(arr),
        "mean":    arr.mean(),
        "median":  np.percentile(arr, 50),
        "p95":     np.percentile(arr, 95),
        "p99":     np.percentile(arr, 99),
        "p999":    np.percentile(arr, 99.9),
        "max":     arr.max(),
    }

def print_report(label: str, stats: dict) -> None:
    print(f"\n── {label} ──")
    for k, v in stats.items():
        print(f"  {k:8s}: {v:8.1f} ms" if k != "n" else f"  {k:8s}: {v}")

if __name__ == "__main__":
    # Scénario 1 : 95% de requêtes rapides, 5% très lentes
    latencies_1 = generate_latencies(
        n_normal=950, n_slow=50,
        normal_ms=50, slow_ms=2000
    )
    print_report("Scénario 1 : 950 requêtes @50ms + 50 @2000ms",
                 compute_stats(latencies_1))

    # Scénario 2 : distribution uniforme entre 10-200ms
    latencies_2 = [random.uniform(10, 200) for _ in range(1000)]
    print_report("Scénario 2 : 1000 requêtes uniformes [10-200ms]",
                 compute_stats(latencies_2))

    # Scénario 3 : ton API FastAPI — modèle réaliste
    # 80% rapides, 15% moyennes, 5% très lentes (GC pause, DB lente...)
    latencies_3 = generate_latencies(800, 150, 30, 300)
    latencies_3 += [random.gauss(1500, 200) for _ in range(50)]
    print_report("Scénario 3 : Distribution trimodale (API réaliste)",
                 compute_stats(latencies_3))
```

### Ce qu'on apprend

La **moyenne** est 147ms — ce qui *paraît* raisonnable. En réalité, **95% des utilisateurs voient 50ms** et **5% voient 2 secondes**. Si tu SLA sur la moyenne, tu rates 50 requêtes/seconde (5% × 1000 req/s) d'utilisateurs avec une latence inacceptable.

Le **p99** (et p999) révèle l'expérience de tes utilisateurs les plus lents — souvent ceux avec le plus de données et donc les plus fidèles ou les plus importants.

**Application directe :** dans ton API FastAPI, si tu logs seulement la latence moyenne, tu *manques* les pannes silencieuses.

```python
# Ajoute dans percentiles.py : simulation de tail latency amplification
# Une opération qui appelle 100 services en parallèle,
# chacun avec p99=200ms — quelle est la latence de l'opération globale ?
import numpy as np

def tail_latency_amplification(n_services: int, percentile_ms: float,
                                n_simulations: int = 10_000) -> float:
    """La latence globale = max des latences des services parallèles."""
    results = []
    for _ in range(n_simulations):
        # Chaque service suit une distribution avec ce p99
        service_latencies = np.random.exponential(percentile_ms / 4.6, n_services)
        results.append(max(service_latencies))  # latence = maillon le plus lent
    return np.percentile(results, 50)  # médiane de la latence globale

for n in [1, 10, 50, 100]:
    lat = tail_latency_amplification(n, 200)
    print(f"  {n:3d} services parallèles → latence médiane globale : {lat:.0f}ms")
```
