### Crée `code/01-storage-retrieval/bitcask.py`

```python
"""
Moteur de stockage Bitcask : log append-only + index hachage.
Inspiration : moteur de stockage de Riak Bitcask (2009).
Simplifié à des fins pédagogiques.
"""
import os
import struct
import time
from pathlib import Path


# Format d'une entrée : [key_len:2B][val_len:4B][timestamp:8B][key][value]
HEADER_FORMAT = ">HI8s"
HEADER_SIZE = struct.calcsize(HEADER_FORMAT)
TOMBSTONE = b"__DELETED__"


class Bitcask:
    def __init__(self, data_dir: str = "/tmp/bitcask"):
        self.data_dir = Path(data_dir)
        self.data_dir.mkdir(parents=True, exist_ok=True)
        self.log_path = self.data_dir / "data.log"
        self.index: dict[str, int] = {}  # clé → byte_offset dans le log
        self._rebuild_index()

    def _rebuild_index(self) -> None:
        """Reconstruit l'index hachage depuis le log (nécessaire après crash)."""
        if not self.log_path.exists():
            return
        with open(self.log_path, "rb") as f:
            offset = 0
            while True:
                header_bytes = f.read(HEADER_SIZE)
                if not header_bytes:
                    break
                key_len, val_len, _ = struct.unpack(HEADER_FORMAT, header_bytes)
                key = f.read(key_len).decode()
                value = f.read(val_len)
                if value == TOMBSTONE:
                    self.index.pop(key, None)
                else:
                    self.index[key] = offset
                offset += HEADER_SIZE + key_len + val_len

    def put(self, key: str, value: str) -> int:
        """Écrit une entrée dans le log. Retourne le byte offset."""
        key_b = key.encode()
        val_b = value.encode()
        ts = struct.pack(">Q", int(time.time() * 1000))  # timestamp en ms
        header = struct.pack(HEADER_FORMAT, len(key_b), len(val_b), ts)
        entry = header + key_b + val_b
        with open(self.log_path, "ab") as f:
            offset = f.tell()
            f.write(entry)
        self.index[key] = offset
        return offset

    def get(self, key: str) -> str | None:
        """Lit une entrée depuis le log via l'index. O(1)."""
        offset = self.index.get(key)
        if offset is None:
            return None
        with open(self.log_path, "rb") as f:
            f.seek(offset)
            header_bytes = f.read(HEADER_SIZE)
            key_len, val_len, _ = struct.unpack(HEADER_FORMAT, header_bytes)
            f.read(key_len)  # skip the key
            return f.read(val_len).decode()

    def delete(self, key: str) -> None:
        """Écrit un tombstone (marqueur de suppression)."""
        key_b = key.encode()
        ts = struct.pack(">Q", int(time.time() * 1000))
        header = struct.pack(HEADER_FORMAT, len(key_b), len(TOMBSTONE), ts)
        entry = header + key_b + TOMBSTONE
        with open(self.log_path, "ab") as f:
            f.write(entry)
        self.index.pop(key, None)

    def compact(self) -> int:
        """Compaction : réécrit le log en gardant seulement la dernière version."""
        tmp_path = self.data_dir / "data.log.tmp"
        new_index: dict[str, int] = {}
        with open(tmp_path, "wb") as out:
            for key in self.index:
                value = self.get(key)
                if value is None:
                    continue
                key_b, val_b = key.encode(), value.encode()
                ts = struct.pack(">Q", int(time.time() * 1000))
                header = struct.pack(HEADER_FORMAT, len(key_b), len(val_b), ts)
                offset = out.tell()
                out.write(header + key_b + val_b)
                new_index[key] = offset
        old_size = self.log_path.stat().st_size if self.log_path.exists() else 0
        tmp_path.replace(self.log_path)
        self.index = new_index
        new_size = self.log_path.stat().st_size
        return old_size - new_size  # octets récupérés

    def stats(self) -> dict:
        log_size = self.log_path.stat().st_size if self.log_path.exists() else 0
        return {
            "clés_en_index": len(self.index),
            "taille_log_octets": log_size,
            "ratio_compaction": log_size / max(len(self.index), 1),
        }


if __name__ == "__main__":
    import shutil
    shutil.rmtree("/tmp/bitcask", ignore_errors=True)

    db = Bitcask()

    # Écriture de base
    print("── Écriture ──")
    db.put("user:1", "Alice")
    db.put("user:2", "Bob")
    db.put("user:3", "Charlie")
    print(f"  get(user:1) → {db.get('user:1')}")
    print(f"  get(user:9) → {db.get('user:9')}")  # None

    # Mise à jour (ancienne valeur reste dans le log !)
    print("\n── Mise à jour ──")
    db.put("user:1", "Alicia")
    print(f"  get(user:1) → {db.get('user:1')}")   # Alicia
    print(f"  stats : {db.stats()}")

    # Avant compaction : le log contient "Alice" et "Alicia"
    before = db.stats()["taille_log_octets"]
    recovered = db.compact()
    after = db.stats()["taille_log_octets"]
    print(f"\n── Compaction ──")
    print(f"  Avant : {before} octets")
    print(f"  Après : {after} octets")
    print(f"  Récupérés : {recovered} octets")
    print(f"  get(user:1) après compaction → {db.get('user:1')}")  # Alicia

    # Suppression
    db.delete("user:2")
    print(f"\n── Suppression ──")
    print(f"  get(user:2) → {db.get('user:2')}")   # None

    # Simulation de redémarrage
    print("\n── Redémarrage (reconstruction de l'index) ──")
    db2 = Bitcask()
    print(f"  get(user:1) après restart → {db2.get('user:1')}")  # Alicia
    print(f"  get(user:2) après restart → {db2.get('user:2')}")  # None (supprimé)
    print(f"  Clés en index : {list(db2.index.keys())}")
```

### Attendu
```
── Écriture ──
  get(user:1) → Alice
  get(user:9) → None

── Mise à jour ──
  get(user:1) → Alicia
  stats : {'clés_en_index': 3, 'taille_log_octets': ~200, ...}

── Compaction ──
  Avant : ~200 octets
  Après : ~150 octets  ← "Alice" supprimé
  Récupérés : ~50 octets

── Redémarrage ──
  get(user:1) → Alicia   ← index reconstruit depuis le log
  get(user:2) → None
```

### Ce qu'on apprend
- **Écriture append-only** = toujours séquentielle = très rapide
- **Index en RAM** = lecture O(1) mais toutes les clés doivent tenir en mémoire
- **Compaction** = trade-off : récupère de l'espace mais consomme I/O
- **Reconstruction d'index** = nécessaire après crash (ou utiliser un WAL séparé)
