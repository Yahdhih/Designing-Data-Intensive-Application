### Étape 1 — Partitionnement par plage (RANGE) sur date (15 min)

```sql
-- Créer une table partitionnée par plage de dates
CREATE TABLE events (
    event_id   BIGSERIAL,
    event_date DATE        NOT NULL,
    user_id    INT,
    action     TEXT,
    payload    JSONB
) PARTITION BY RANGE (event_date);

-- Créer les partitions trimestrielles (Q1 2024, Q2 2024, Q3 2024)
CREATE TABLE events_2024_q1 PARTITION OF events
    FOR VALUES FROM ('2024-01-01') TO ('2024-04-01');

CREATE TABLE events_2024_q2 PARTITION OF events
    FOR VALUES FROM ('2024-04-01') TO ('2024-07-01');

CREATE TABLE events_2024_q3 PARTITION OF events
    FOR VALUES FROM ('2024-07-01') TO ('2024-10-01');

-- Partition par défaut (pour les dates hors plage définies)
CREATE TABLE events_default PARTITION OF events DEFAULT;

-- Insérer des données dans plusieurs trimestres
INSERT INTO events (event_date, user_id, action, payload)
SELECT
    ('2024-01-01'::DATE + (RANDOM() * 270)::INT) AS event_date,
    (RANDOM() * 10000)::INT AS user_id,
    CASE (RANDOM() * 3)::INT
        WHEN 0 THEN 'login'
        WHEN 1 THEN 'purchase'
        ELSE 'logout'
    END AS action,
    jsonb_build_object('ip', '192.168.1.' || (RANDOM() * 255)::INT)
FROM generate_series(1, 100000);

-- Vérifier la distribution par partition
SELECT
    tableoid::regclass AS partition_name,
    COUNT(*) AS rows
FROM events
GROUP BY tableoid
ORDER BY partition_name;
```

### Étape 2 — Observer le partition pruning (10 min)

```sql
-- Requête sur 1 trimestre → Postgres ne lit que la partition concernée
EXPLAIN ANALYZE
SELECT COUNT(*) FROM events
WHERE event_date BETWEEN '2024-04-01' AND '2024-06-30';

-- Dans la sortie EXPLAIN : cherche "Partitions: events_2024_q2"
-- → Postgres a éliminé les autres partitions (partition pruning) ✓

-- Requête SANS clause de partition → Postgres doit lire toutes les partitions
EXPLAIN ANALYZE
SELECT COUNT(*) FROM events
WHERE action = 'purchase';
-- → Cherche "Partitions: events_2024_q1, events_2024_q2, events_2024_q3, events_default"
-- → Toutes les partitions sont interrogées (equivalent to scatter/gather)
```

### Étape 3 — Partitionnement par hachage (HASH) (10 min)

```sql
-- Table partitionnée par hachage (distribution uniforme)
CREATE TABLE users_sharded (
    user_id    INT NOT NULL,
    username   TEXT,
    email      TEXT,
    country    TEXT
) PARTITION BY HASH (user_id);

-- 4 shards hash
CREATE TABLE users_shard_0 PARTITION OF users_sharded
    FOR VALUES WITH (MODULUS 4, REMAINDER 0);
CREATE TABLE users_shard_1 PARTITION OF users_sharded
    FOR VALUES WITH (MODULUS 4, REMAINDER 1);
CREATE TABLE users_shard_2 PARTITION OF users_sharded
    FOR VALUES WITH (MODULUS 4, REMAINDER 2);
CREATE TABLE users_shard_3 PARTITION OF users_sharded
    FOR VALUES WITH (MODULUS 4, REMAINDER 3);

-- Insérer des utilisateurs
INSERT INTO users_sharded (user_id, username, email, country)
SELECT
    i AS user_id,
    'user_' || i AS username,
    'user_' || i || '@example.com' AS email,
    CASE (i % 5)
        WHEN 0 THEN 'France'
        WHEN 1 THEN 'Germany'
        WHEN 2 THEN 'Spain'
        WHEN 3 THEN 'Italy'
        ELSE 'Other'
    END AS country
FROM generate_series(1, 100000) i;

-- Distribution des shards
SELECT tableoid::regclass AS shard, COUNT(*) FROM users_sharded GROUP BY tableoid ORDER BY shard;

-- Requête par clé primaire → partition pruning ✓
EXPLAIN ANALYZE SELECT * FROM users_sharded WHERE user_id = 42;

-- Requête par attribut secondaire → scatter sur tous les shards ✗
EXPLAIN ANALYZE SELECT * FROM users_sharded WHERE country = 'France';
```

**Note :** La requête `WHERE country = 'France'` avec partitionnement HASH sur `user_id` nécessite d'interroger les 4 shards — c'est exactement le problème de l'index secondaire vu en J.46.

### Étape 4 — Index sur la table partitionnée (10 min)

```sql
-- Créer un index sur 'country' dans la table partitionnée
-- Postgres crée automatiquement 1 index LOCAL par partition
CREATE INDEX ON users_sharded (country);

-- La requête scatter est-elle accélérée ?
EXPLAIN ANALYZE SELECT * FROM users_sharded WHERE country = 'France';
-- → Elle interroge toujours les 4 shards, mais avec index local (Bitmap Heap Scan)
-- → C'est le "scatter/gather avec index local" de Ch.6 !

-- Nettoyage
DROP TABLE IF EXISTS events CASCADE;
DROP TABLE IF EXISTS users_sharded CASCADE;
```
