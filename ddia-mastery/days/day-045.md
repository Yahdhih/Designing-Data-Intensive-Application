```python
"""
Anneau de hachage cohérent avec virtual nodes (vnodes).
Démontre :
  1. Distribution uniforme des clés (avec vs sans vnodes)
  2. Rééquilibrage minimal lors de l'ajout/suppression d'un nœud
  3. Hot spot avec des clés biaisées (salting comme solution)
"""
import hashlib
import bisect
import math
from collections import defaultdict


# ═══════════════════════════════════════════════════════════════════════════════
# Anneau de hachage cohérent
# ═══════════════════════════════════════════════════════════════════════════════

class ConsistentHashRing:
    """
    Anneau de hachage cohérent.
    Chaque nœud physique est représenté par `n_vnodes` positions sur l'anneau.
    Une clé est assignée au premier nœud dans le sens horaire depuis sa position.
    """

    def __init__(self, n_vnodes: int = 150):
        self.n_vnodes = n_vnodes
        self._ring: list[int] = []            # positions triées sur l'anneau
        self._ring_map: dict[int, str] = {}   # position → nom du nœud physique

    def _hash(self, key: str) -> int:
        """Hash 128 bits → entier (MD5 pour la démo, SHA-256 en production)."""
        return int(hashlib.md5(key.encode()).hexdigest(), 16)

    def add_node(self, node: str) -> set[str]:
        """
        Ajoute un nœud + ses vnodes.
        Retourne l'ensemble des clés qui doivent migrer vers ce nœud
        (pour illustration — dans un vrai système, c'est le nœud qui demande ses données).
        """
        for i in range(self.n_vnodes):
            pos = self._hash(f"{node}#vnode#{i}")
            bisect.insort(self._ring, pos)
            self._ring_map[pos] = node
        return set()  # migration calculée séparément

    def remove_node(self, node: str) -> None:
        """Supprime un nœud et tous ses vnodes."""
        to_remove = [pos for pos, n in self._ring_map.items() if n == node]
        for pos in to_remove:
            self._ring.remove(pos)
            del self._ring_map[pos]

    def get_node(self, key: str) -> str:
        """Retourne le nœud responsable de cette clé."""
        if not self._ring:
            raise ValueError("Aucun nœud dans l'anneau")
        pos = self._hash(key)
        # Chercher la première position >= pos (modulo longueur de l'anneau)
        idx = bisect.bisect_left(self._ring, pos) % len(self._ring)
        return self._ring_map[self._ring[idx]]

    def distribution(self, keys: list[str]) -> dict[str, int]:
        """Compter le nombre de clés assignées à chaque nœud."""
        counts: dict[str, int] = defaultdict(int)
        for k in keys:
            counts[self.get_node(k)] += 1
        return dict(counts)

    def std_dev(self, keys: list[str]) -> float:
        """Écart-type de la distribution (0 = parfaitement uniforme)."""
        dist = self.distribution(keys)
        values = list(dist.values())
        mean = sum(values) / len(values)
        variance = sum((v - mean) ** 2 for v in values) / len(values)
        return math.sqrt(variance)


# ═══════════════════════════════════════════════════════════════════════════════
# Démonstrations
# ═══════════════════════════════════════════════════════════════════════════════

def demo_distribution(keys: list[str]) -> None:
    """Comparer la distribution avec 1 vnode vs 150 vnodes."""
    print("\n═══ Démonstration 1 : Distribution (1 vnode vs 150 vnodes) ═══")
    nodes = ["node-A", "node-B", "node-C"]

    for n_vnodes in [1, 10, 150]:
        ring = ConsistentHashRing(n_vnodes=n_vnodes)
        for node in nodes:
            ring.add_node(node)

        dist = ring.distribution(keys)
        total = len(keys)
        expected = total // len(nodes)
        std = ring.std_dev(keys)

        print(f"\n  {n_vnodes} vnode(s) / nœud :")
        for node in sorted(dist):
            bar = "█" * (dist[node] * 40 // total)
            print(f"    {node}: {dist.get(node, 0):6d} clés  {bar}")
        print(f"    Attendu : ~{expected} par nœud | Écart-type : {std:.0f}")


def demo_rebalancing(keys: list[str]) -> None:
    """Montrer que l'ajout d'un nœud ne déplace qu'une fraction des clés."""
    print("\n═══ Démonstration 2 : Rééquilibrage minimal ═══")
    ring = ConsistentHashRing(n_vnodes=150)
    nodes = ["node-A", "node-B", "node-C"]
    for node in nodes:
        ring.add_node(node)

    # Assignation initiale
    before = {k: ring.get_node(k) for k in keys}

    # Ajouter un nœud
    ring.add_node("node-D")
    after = {k: ring.get_node(k) for k in keys}

    moved = sum(1 for k in keys if before[k] != after[k])
    expected_move = len(keys) // (len(nodes) + 1)

    print(f"  Clés totales : {len(keys)}")
    print(f"  Ajout de node-D (4 nœuds total)")
    print(f"  Clés déplacées : {moved} ({moved * 100 / len(keys):.1f}%)")
    print(f"  Attendu (idéal ~1/N) : ~{expected_move} ({expected_move * 100 / len(keys):.1f}%)")

    # Distribution après ajout
    dist_after = ring.distribution(keys)
    print(f"  Distribution après ajout :")
    for node in sorted(dist_after):
        bar = "█" * (dist_after[node] * 30 // len(keys))
        print(f"    {node}: {dist_after[node]:5d} clés  {bar}")

    # Comparaison avec hash % N (catastrophique)
    moved_mod = sum(1 for k in keys if hash(k) % 3 != hash(k) % 4)
    print(f"\n  Comparaison avec hash(k) % N :")
    print(f"    hash(k) % 3 → hash(k) % 4 : {moved_mod} clés déplacées ({moved_mod * 100 / len(keys):.1f}%)")
    print(f"    → Hachage cohérent est {moved_mod / max(moved, 1):.1f}× moins perturbateur")


def demo_hotspot(keys_skewed: list[str], salt: bool = False) -> None:
    """Montrer un hot spot et la correction par salting."""
    print(f"\n═══ Démonstration 3 : Hot spot {'avec salting' if salt else 'sans salting'} ═══")
    ring = ConsistentHashRing(n_vnodes=150)
    for node in ["node-A", "node-B", "node-C"]:
        ring.add_node(node)

    if salt:
        # Salting : disperser la même clé populaire sur N versions
        salted = []
        for k in keys_skewed:
            if k == "celebrity:999":
                for i in range(10):  # 10 partitions salt
                    salted.append(f"celebrity:999#{i}")
            else:
                salted.append(k)
        effective_keys = salted
    else:
        effective_keys = keys_skewed

    dist = ring.distribution(effective_keys)
    total = len(effective_keys)
    print(f"  Distribution ({total} clés effectives) :")
    for node in sorted(dist):
        bar = "█" * (dist[node] * 40 // total)
        print(f"    {node}: {dist[node]:5d} clés  {bar}")
    if not salt:
        print("  ⚠ Hot spot probable sur 1 nœud (toutes les écritures celebrity:999)")
    else:
        print("  ✓ Salting distribue celebrity:999 sur plusieurs nœuds")


def main() -> None:
    # Générer 10 000 clés "normales"
    keys = [f"user:{i}" for i in range(10_000)]

    # 9 500 clés normales + 500 fois la même clé populaire (hot spot)
    keys_skewed = [f"user:{i}" for i in range(9_500)] + ["celebrity:999"] * 500

    demo_distribution(keys)
    demo_rebalancing(keys)
    demo_hotspot(keys_skewed, salt=False)
    demo_hotspot(keys_skewed, salt=True)


if __name__ == "__main__":
    main()
```
