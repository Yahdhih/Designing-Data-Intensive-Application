### Crée `code/02-encoding/encoder.py`

```python
"""
Encodeur binaire longueur-préfixé.
Format d'un champ : [field_id: 1B][type: 1B][length: 2B][data: N bytes]
Inspiré de Protobuf (très simplifié — pédagogique seulement).

Compatibilité :
- Forward compat : décodeur v1 ignore les champs inconnus (field_id > connu)
- Backward compat : décodeur v2 met None pour les champs absents
"""
import struct

# ── Types supportés ──────────────────────────────────────────────────────────
TYPE_INT    = 0x01
TYPE_STRING = 0x02

# Format d'en-tête d'un champ : field_id (1B) + type (1B) + length (2B)
FIELD_HEADER = ">BBH"
FIELD_HEADER_SIZE = struct.calcsize(FIELD_HEADER)


def encode(data: dict[int, object]) -> bytes:
    """
    Encode un dict {field_id: valeur} en bytes.
    Les clés sont les IDs de champ (entiers), pas les noms.
    """
    result = bytearray()
    for field_id, value in sorted(data.items()):
        if isinstance(value, int):
            payload = struct.pack(">i", value)  # 4 bytes big-endian
            type_tag = TYPE_INT
        elif isinstance(value, str):
            payload = value.encode("utf-8")
            type_tag = TYPE_STRING
        else:
            raise ValueError(f"Type non supporté : {type(value)}")
        header = struct.pack(FIELD_HEADER, field_id, type_tag, len(payload))
        result += header + payload
    return bytes(result)


def decode(data: bytes, schema: dict[int, tuple[str, type]]) -> dict[str, object]:
    """
    Décode des bytes avec un schéma {field_id: (nom, type)}.
    Champs inconnus → ignorés (forward compat).
    Champs absents → None (backward compat si le champ était optionnel).
    """
    result = {name: None for _, (name, _) in schema.items()}
    offset = 0
    while offset < len(data):
        if offset + FIELD_HEADER_SIZE > len(data):
            break
        field_id, type_tag, length = struct.unpack_from(FIELD_HEADER, data, offset)
        offset += FIELD_HEADER_SIZE
        payload = data[offset:offset + length]
        offset += length

        if field_id in schema:
            name, _ = schema[field_id]
            if type_tag == TYPE_INT:
                result[name] = struct.unpack(">i", payload)[0]
            elif type_tag == TYPE_STRING:
                result[name] = payload.decode("utf-8")
        # Sinon : field_id inconnu → ignoré (forward compat automatique)

    return result


# ── Schémas ──────────────────────────────────────────────────────────────────

# Schéma v1 : {user_id: int, name: str}
SCHEMA_V1 = {
    1: ("user_id", int),
    2: ("name", str),
}

# Schéma v2 : {user_id: int, name: str, email: str}  ← champ 3 ajouté
SCHEMA_V2 = {
    1: ("user_id", int),
    2: ("name", str),
    3: ("email", str),      # NOUVEAU champ optionnel
}

# Schéma v3 : {user_id: int, name: str, email: str, score: int}  ← champ 4 ajouté
SCHEMA_V3 = {
    1: ("user_id", int),
    2: ("name", str),
    3: ("email", str),
    4: ("score", int),      # NOUVEAU champ optionnel
}


if __name__ == "__main__":
    import json

    # ── Test de base ──────────────────────────────────────────────────────────
    print("═══ Encodage/décodage basique ═══")
    msg_v1 = encode({1: 42, 2: "Alice"})
    decoded = decode(msg_v1, SCHEMA_V1)
    print(f"  encode({{1: 42, 2: 'Alice'}}) = {msg_v1.hex()}")
    print(f"  decode avec schéma v1 = {decoded}")

    json_equiv = json.dumps({"user_id": 42, "name": "Alice"}).encode()
    print(f"\n  Taille JSON   : {len(json_equiv)} bytes → '{json_equiv.decode()}'")
    print(f"  Taille binaire: {len(msg_v1)} bytes → {msg_v1.hex()}")
    print(f"  Ratio         : {len(json_equiv) / len(msg_v1):.1f}× plus compact")

    # ── Forward compatibility ─────────────────────────────────────────────────
    print("\n═══ Forward compatibility : vieux décodeur (v1) lit nouveau message (v2) ═══")
    msg_v2 = encode({1: 42, 2: "Alice", 3: "alice@example.com"})
    decoded_by_v1 = decode(msg_v2, SCHEMA_V1)
    print(f"  Message v2 encodé : {msg_v2.hex()}")
    print(f"  Décodé par v1 : {decoded_by_v1}")
    print(f"  → Le champ 'email' (field_id=3) est IGNORÉ silencieusement ✓")
    assert decoded_by_v1["user_id"] == 42
    assert decoded_by_v1["name"] == "Alice"
    assert "email" not in decoded_by_v1

    # ── Backward compatibility ────────────────────────────────────────────────
    print("\n═══ Backward compatibility : nouveau décodeur (v2) lit vieux message (v1) ═══")
    decoded_by_v2 = decode(msg_v1, SCHEMA_V2)
    print(f"  Message v1 décodé par v2 : {decoded_by_v2}")
    print(f"  → Le champ 'email' absent → None (valeur par défaut) ✓")
    assert decoded_by_v2["user_id"] == 42
    assert decoded_by_v2["name"] == "Alice"
    assert decoded_by_v2["email"] is None

    # ── Danger : changer le numéro de champ ──────────────────────────────────
    print("\n═══ DANGER : changer le numéro de champ = incompatibilité ═══")
    # Si quelqu'un réassigne field_id=2 à "email" au lieu de "name"
    BAD_SCHEMA = {
        1: ("user_id", int),
        2: ("email", str),   # ← mauvaise idée : on a réutilisé field_id=2 !
    }
    decoded_bad = decode(msg_v1, BAD_SCHEMA)
    print(f"  Message v1 ('name'='Alice') décodé avec schéma corrompu :")
    print(f"  → {decoded_bad}")
    print(f"  → 'email' = 'Alice' !? Données silencieusement corrompues ✗")

    print("\n✅ Tous les tests de compatibilité passés.")
```
