```python
"""
Démonstration des 3 anomalies de concurrence que Snapshot Isolation laisse passer.
Chaque anomalie est montrée avec et sans solution.
"""
import sys
import os
sys.path.insert(0, os.path.dirname(__file__))
from mvcc_store import MVCCStore


# ═══════════════════════════════════════════════════════════════════════════════
# Anomalie 1 : Lost Update
# ═══════════════════════════════════════════════════════════════════════════════

def demo_lost_update_probleme() -> None:
    """Read-modify-write concurrent → un des 2 incréments est perdu."""
    print("\n═══ Anomalie 1 : Lost Update (PROBLÈME) ═══")
    store = MVCCStore()
    t0 = store.begin()
    t0.write("page_views", "1000")
    t0.commit()

    # T1 et T2 lisent, incrémentent, écrivent
    t1 = store.begin()
    t2 = store.begin()

    v1 = t1.read("page_views")
    v2 = t2.read("page_views")
    print(f"  T1 lit page_views → {v1}")
    print(f"  T2 lit page_views → {v2}")

    t1.write("page_views", str(int(v1) + 1))
    t1.commit()

    t2.write("page_views", str(int(v2) + 1))
    t2.commit()

    t3 = store.begin()
    final = t3.read("page_views")
    t3.commit()
    print(f"  Résultat : page_views = {final} (attendu : 1002)")
    print(f"  ✗ Lost update ! L'incrément de T1 a été écrasé par T2.")


def demo_lost_update_solution() -> None:
    """Solution : atomic write (pas de read-modify-write)."""
    print("\n  Solution : utiliser une opération atomique côté BDD")
    print("  SQL : UPDATE stats SET page_views = page_views + 1 WHERE id = 1")
    print("  → L'incrément est atomique au niveau du moteur de stockage (sans lire en Python)")
    print("  → Équivalent dans notre store : non applicable (nécessite support BDD)")
    print("  → Ou bien : SELECT FOR UPDATE (verrou exclusif avant lecture)")


# ═══════════════════════════════════════════════════════════════════════════════
# Anomalie 2 : Write Skew (le problème des médecins de garde)
# ═══════════════════════════════════════════════════════════════════════════════

def demo_write_skew_probleme() -> None:
    """
    Invariant : au moins 1 médecin disponible.
    Les 2 lisent '2 disponibles', les 2 se libèrent → 0 disponible.
    """
    print("\n═══ Anomalie 2 : Write Skew — médecins de garde (PROBLÈME) ═══")
    store = MVCCStore()

    # Initialiser : Alice et Bob sont disponibles
    t0 = store.begin()
    t0.write("doctor:alice:on_call", "true")
    t0.write("doctor:bob:on_call", "true")
    t0.commit()

    def count_on_call(txn) -> int:
        alice = txn.read("doctor:alice:on_call")
        bob = txn.read("doctor:bob:on_call")
        return sum(1 for x in [alice, bob] if x == "true")

    # T1 (Alice) et T2 (Bob) commencent simultanément
    t_alice = store.begin()
    t_bob = store.begin()

    n_alice = count_on_call(t_alice)
    n_bob = count_on_call(t_bob)
    print(f"  Alice voit {n_alice} médecins disponibles → se libère")
    print(f"  Bob voit {n_bob} médecins disponibles → se libère")

    t_alice.write("doctor:alice:on_call", "false")
    t_alice.commit()

    t_bob.write("doctor:bob:on_call", "false")
    t_bob.commit()

    # Vérifier l'état final
    t_check = store.begin()
    alice_final = t_check.read("doctor:alice:on_call")
    bob_final = t_check.read("doctor:bob:on_call")
    t_check.commit()

    n_final = sum(1 for x in [alice_final, bob_final] if x == "true")
    print(f"  État final : Alice={alice_final}, Bob={bob_final} → {n_final} médecin(s) disponible(s)")
    print(f"  ✗ Write skew ! Invariant violé : {n_final} médecin(s) disponible(s) (requis : ≥ 1)")


def demo_write_skew_solution() -> None:
    """
    Solution : matérialisé le conflit avec un verrou explicite.
    Dans notre store simplifié, on simule SELECT FOR UPDATE.
    En SQL : SELECT COUNT(*) FROM doctors WHERE on_call = true FOR UPDATE;
    """
    print("\n  Solution : SELECT FOR UPDATE sur les lignes lues")
    print("  SQL réel :")
    print("    BEGIN;")
    print("    SELECT COUNT(*) FROM doctors WHERE on_call = true FOR UPDATE;")
    print("    -- Si count > 1, mettre à jour. Sinon ROLLBACK.")
    print("    UPDATE doctors SET on_call = false WHERE name = 'Alice';")
    print("    COMMIT;")
    print("  → T2 doit attendre que T1 ait terminé avant de lire le count")
    print("  → Bob voit alors : 1 médecin disponible → ne peut pas se libérer ✓")


# ═══════════════════════════════════════════════════════════════════════════════
# Anomalie 3 : Phantom Read (double réservation de salle)
# ═══════════════════════════════════════════════════════════════════════════════

def demo_phantom_read() -> None:
    """
    T1 vérifie qu'une salle est libre → T2 la réserve → T1 la réserve aussi.
    Résultat : double réservation.
    """
    print("\n═══ Anomalie 3 : Phantom Read — double réservation (PROBLÈME) ═══")
    store = MVCCStore()

    # Pas de réservation existante
    t0 = store.begin()
    t0.write("bookings:count:room123:dec25", "0")
    t0.commit()

    # T1 (Client A) vérifie la disponibilité
    t1 = store.begin()
    existing = t1.read("bookings:count:room123:dec25")
    print(f"  Client A voit {existing} réservation(s) pour la chambre 123 le 25 déc → libre !")

    # T2 (Client B) réserve la même chambre
    t2 = store.begin()
    existing_b = t2.read("bookings:count:room123:dec25")
    print(f"  Client B voit {existing_b} réservation(s) → réserve !")
    t2.write("bookings:count:room123:dec25", "1")
    t2.write("bookings:room123:dec25:owner", "Client-B")
    t2.commit()
    print(f"  Client B a réservé (et commité)")

    # T1 réserve aussi (son snapshot ne voit pas T2)
    t1.write("bookings:count:room123:dec25", "1")  # ← écrase la valeur de T2 !
    t1.write("bookings:room123:dec25:owner", "Client-A")
    t1.commit()
    print(f"  Client A a aussi réservé (son snapshot = avant T2)")

    # Vérifier l'état final : qui a la chambre ?
    t3 = store.begin()
    owner = t3.read("bookings:room123:dec25:owner")
    count = t3.read("bookings:count:room123:dec25")
    t3.commit()
    print(f"  Gagnant MVCC : {owner} (count={count})")
    print(f"  ✗ Phantom read ! Les 2 clients pensaient avoir réservé.")
    print(f"  → Solution : Serializable (prédicat lock sur la plage de dates)")

    print("\n  Solution SQL :")
    print("    BEGIN ISOLATION LEVEL SERIALIZABLE;")
    print("    SELECT * FROM bookings WHERE room=123 AND date='2024-12-25' FOR UPDATE;")
    print("    -- Si 0 résultats : INSERT INTO bookings...")
    print("    COMMIT;")


# ═══════════════════════════════════════════════════════════════════════════════
# Récapitulatif
# ═══════════════════════════════════════════════════════════════════════════════

def print_summary() -> None:
    print("\n═══ Récapitulatif des anomalies et solutions ═══")
    rows = [
        ("Lost update",   "Snapshot Isolation", "Atomic write / SELECT FOR UPDATE / CAS"),
        ("Write skew",    "Snapshot Isolation", "SELECT FOR UPDATE + lock toutes lignes lues"),
        ("Phantom read",  "Snapshot Isolation", "Serializable (predicate lock ou SSI)"),
    ]
    print(f"  {'Anomalie':<20} {'SI empêche?':<22} {'Solution'}")
    print(f"  {'-'*70}")
    for anomaly, si, solution in rows:
        print(f"  {anomaly:<20} {'✗ Non':<22} {solution}")


def main() -> None:
    demo_lost_update_probleme()
    demo_lost_update_solution()
    demo_write_skew_probleme()
    demo_write_skew_solution()
    demo_phantom_read()
    print_summary()


if __name__ == "__main__":
    main()
```
