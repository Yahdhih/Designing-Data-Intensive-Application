### Objectif
Implémenter un B-tree simplifié (B+ tree en réalité) :
1. Structure : nœuds avec clés + enfants (degree = 3)
2. Recherche d'une clé O(log n)
3. Insertion avec split de nœud quand plein
4. Traversée en ordre (range scan)

### Crée `code/01-storage-retrieval/btree.py`

```python
"""
Mini-B-tree (degree=3, ordre 2-3).
Simplifié : pas de disque, pas de WAL, clés entières pour la lisibilité.
Objectif : comprendre la structure et le split.
"""
from __future__ import annotations
from dataclasses import dataclass, field


@dataclass
class BTreeNode:
    degree: int = 3         # max d'enfants par nœud
    keys: list[int] = field(default_factory=list)
    values: list[str] = field(default_factory=list)
    children: list["BTreeNode"] = field(default_factory=list)

    @property
    def is_leaf(self) -> bool:
        return len(self.children) == 0

    @property
    def is_full(self) -> bool:
        return len(self.keys) >= self.degree - 1  # max degree-1 clés

    def __repr__(self) -> str:
        return f"Node(keys={self.keys})"


class BTree:
    def __init__(self, degree: int = 3):
        self.degree = degree
        self.root = BTreeNode(degree=degree)

    # ── Recherche ─────────────────────────────────────────────────────────────

    def search(self, key: int, node: BTreeNode | None = None) -> str | None:
        """Recherche récursive — O(log n) par descente."""
        if node is None:
            node = self.root
        # Trouver la position de la clé dans le nœud
        i = 0
        while i < len(node.keys) and key > node.keys[i]:
            i += 1
        if i < len(node.keys) and key == node.keys[i]:
            return node.values[i]
        if node.is_leaf:
            return None  # pas trouvé
        return self.search(key, node.children[i])

    # ── Insertion ─────────────────────────────────────────────────────────────

    def insert(self, key: int, value: str) -> None:
        root = self.root
        if root.is_full:
            # La racine est pleine : crée une nouvelle racine et split
            new_root = BTreeNode(degree=self.degree)
            new_root.children.append(self.root)
            self._split_child(new_root, 0)
            self.root = new_root
        self._insert_non_full(self.root, key, value)

    def _insert_non_full(self, node: BTreeNode, key: int, value: str) -> None:
        i = len(node.keys) - 1
        if node.is_leaf:
            # Insérer la clé à la bonne position (ordre trié)
            node.keys.append(0)
            node.values.append("")
            while i >= 0 and key < node.keys[i]:
                node.keys[i + 1] = node.keys[i]
                node.values[i + 1] = node.values[i]
                i -= 1
            node.keys[i + 1] = key
            node.values[i + 1] = value
        else:
            # Descendre dans le bon enfant
            while i >= 0 and key < node.keys[i]:
                i -= 1
            i += 1
            if node.children[i].is_full:
                self._split_child(node, i)
                if key > node.keys[i]:
                    i += 1
            self._insert_non_full(node.children[i], key, value)

    def _split_child(self, parent: BTreeNode, i: int) -> None:
        """Split le i-ème enfant de parent (qui est plein)."""
        t = self.degree // 2  # degré minimum
        child = parent.children[i]
        new_child = BTreeNode(degree=self.degree)

        # La clé médiane monte vers le parent
        mid = len(child.keys) // 2
        median_key = child.keys[mid]
        median_val = child.values[mid]

        # Le nouveau nœud prend la moitié droite
        new_child.keys = child.keys[mid + 1:]
        new_child.values = child.values[mid + 1:]
        if not child.is_leaf:
            new_child.children = child.children[mid + 1:]
            child.children = child.children[:mid + 1]

        # L'enfant original garde la moitié gauche
        child.keys = child.keys[:mid]
        child.values = child.values[:mid]

        # Insérer la médiane dans le parent
        parent.keys.insert(i, median_key)
        parent.values.insert(i, median_val)
        parent.children.insert(i + 1, new_child)

    # ── Traversée en ordre ────────────────────────────────────────────────────

    def range_scan(self, lo: int, hi: int) -> list[tuple[int, str]]:
        """Retourne toutes les paires (clé, valeur) dans [lo, hi]."""
        result = []
        self._range_scan(self.root, lo, hi, result)
        return result

    def _range_scan(self, node: BTreeNode, lo: int, hi: int,
                    result: list[tuple[int, str]]) -> None:
        for i, key in enumerate(node.keys):
            if not node.is_leaf:
                self._range_scan(node.children[i], lo, hi, result)
            if lo <= key <= hi:
                result.append((key, node.values[i]))
        if not node.is_leaf:
            self._range_scan(node.children[-1], lo, hi, result)

    def height(self, node: BTreeNode | None = None) -> int:
        if node is None:
            node = self.root
        if node.is_leaf:
            return 1
        return 1 + self.height(node.children[0])

    def __repr__(self) -> str:
        return f"BTree(height={self.height()}, root={self.root})"


if __name__ == "__main__":
    import random

    bt = BTree(degree=4)  # max 3 clés par nœud

    # Insérer 20 clés aléatoires
    keys = random.sample(range(1, 101), 20)
    print(f"Insertion de : {sorted(keys)}")
    for k in keys:
        bt.insert(k, f"val_{k}")

    print(f"\nAprès insertion : {bt}")
    print(f"  Hauteur de l'arbre : {bt.height()}")

    # Recherche
    for k in [keys[0], keys[-1], 999]:
        result = bt.search(k)
        print(f"  search({k}) → {result}")

    # Range scan
    lo, hi = min(keys) + 5, max(keys) - 5
    if lo < hi:
        found = bt.range_scan(lo, hi)
        print(f"\nRange scan [{lo}, {hi}] → {[k for k,_ in found]}")

    # Vérification : toutes les clés insérées doivent être trouvables
    print(f"\nVérification : toutes les clés retrouvables ?")
    all_found = all(bt.search(k) is not None for k in keys)
    print(f"  → {'✓ OUI' if all_found else '✗ NON — BUG !'}")

    # Comparaison hauteur avec un arbre binaire
    print(f"\nComparaison :")
    print(f"  B-tree (degree=4) avec 20 clés : hauteur = {bt.height()}")
    print(f"  Arbre binaire parfait avec 20 clés : hauteur = ~{20 .bit_length()}")
    print(f"  (B-tree est plus plat → moins d'I/O disque)")
```

### Ce qu'on apprend
- Le B-tree est plus plat qu'un arbre binaire → moins de pages lues par recherche
- Le split de nœud est l'opération clé → avec WAL en production (Postgres)
- Le range scan fonctionne naturellement (clés triées dans les nœuds)
- En production, chaque nœud = 1 page de 4KB → lit/écrit une page entière par niveau
